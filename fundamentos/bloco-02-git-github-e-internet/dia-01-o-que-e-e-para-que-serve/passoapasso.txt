//Instalação do Git (via comandos no terminal)
sudo apt-get install git-all

//Configuração do Git
git config --global user.name "Evandro Valente"
git config --global user.email evandr.valente@gmail.com

//Configurar um editor para poder acessar e alterar o arquivo de configuração do Git , .gitconfig
git config --global core.editor "code --wait"

//Verificar satus da instalação e configuração
git --version 
git config --list

//Gerando uma chave SSH para utilizar com  minha conta do GitHub
ssh-keygen -t rsa -b 4096 -C "evandr.valente@gmail.com"

//Adicionando sua chave SSH ao ssh-agent
eval "$(ssh-agent -s)"
ssh-add ~/.ssh/id_rsa

//Copiando a chave gerada para utilização no GitHub
xclip -sel clip < ~/.ssh/id_rsa.pub

//Adicionando a chave SSH gerada na minha conta do GitHub (via navegador)
Entrar no GitHub
No canto superior direito do GitHub , entrar em Settings ;
Na barra lateral esquerda, clicar em SSH and GPG keys ;
Clicar em New SSH key ou Add SSH key ;
No campo Título , adicionar uma descrição para a nova chave;
Colar  dentro do campo Key a chave que foi gerada;
Clicar em Add SSH key ;
